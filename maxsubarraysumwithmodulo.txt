#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;


int main() {
    long long c,i,j,k,l,m,n;
    cin>>c;
    while(c>0){
        cin>>n;cin>>m;
        long long min,maxx;
        pair<long long,long long> a[n];
        for(i=0;i<n;i++)
            {
            cin>>a[i].first;
            a[i].second=i+1;
        }
        //calculating prefix sum
        a[0].first=(a[0].first)%m;
        maxx=a[0].first;
        for(i=1;i<n;i++)
            {  a[i].first=((a[i].first)%m+(a[i-1].first)%m)%m;
              if(maxx<a[i].first)
            maxx=a[i].first;
            
        }
        
        sort(a,a+n);
        min=LONG_MAX;
        for(i=0;i<n-1;i++)
            {
            if(a[i].second > a[i+1].second)
                {
                if((a[i+1].first-a[i].first)<min)
                    {
                    min=a[i+1].first-a[i].first;
                }
            }
            
        }
            cout<<max(maxx,m-min)<<"\n";
        
        
        
        
        
        
     c--;   
    }
    
    
    
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
